\newcommand{\fDFA}[3]{
	\delta_D(\{
	\foreach[count=\i] \j in {#1} {
		\ifthenelse{\i < #2}{s_\j, }{s_\j} 
	}
	\}, #3)
}

\newcommand{\fNFA}[2]{
	\delta_N(s_{#1}, #2)
}

\newcommand{\convNFAtoDFA}[5]{
	$\fDFA{#1}{#2}{#3} = 
	\foreach[count=\i] \j in {#1} {
		\fNFA{\j}{#3}
		\ifthenelse{\i < #2}{\cup}{}
	} = \{
	\foreach[count=\i] \j in {#4} {
		\ifthenelse{\i < #5}{s_\j, }{s_\j}
	}\}$
}
